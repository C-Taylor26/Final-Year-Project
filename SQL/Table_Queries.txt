---DATA TABLE---
CREATE TABLE `ai-trader-data`.`data` ( `ID` INT(3) NOT NULL AUTO_INCREMENT , `Date` DATE NOT NULL , `Stock` VARCHAR(4) NOT NULL , `Open` DECIMAL(4,2) NOT NULL , `Close` DECIMAL(4,2) NOT NULL , `20MA_Open` DECIMAL(4,2) NOT NULL , `50MA_Open` DECIMAL(4,2) NOT NULL , `200MA_Open` DECIMAL(4,2) NOT NULL , `20MA_Close` DECIMAL(4,2) NOT NULL , `50MA_Close` DECIMAL(4,2) NOT NULL , `200MA_Close` DECIMAL(4,2) NOT NULL , `Next_Day_Change` DECIMAL(4,2) NULL , PRIMARY KEY (`ID`)) ENGINE = MyISAM;

---TRADES TABLE---
CREATE TABLE `ai-trader-data`.`trades` ( `ID` INT(6) NOT NULL AUTO_INCREMENT , `DateTime` TIMESTAMP NOT NULL , `Stock` VARCHAR(4) NOT NULL , `Open` DECIMAL(4,2) NOT NULL , `Close` DECIMAL(4,2) NOT NULL , `Direction` VARCHAR(4) NOT NULL , `Indicator` FLOAT NULL , PRIMARY KEY (`ID`)) ENGINE = MyISAM;
